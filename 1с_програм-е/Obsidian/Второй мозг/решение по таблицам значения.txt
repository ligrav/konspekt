Процедура ВыполнитьКодНаСервере()
	
	//в этой процедуре напишите программный код, который решает задачу
	ТаблицаТовары = Новый ТаблицаЗначений;
	
	ТаблицаТовары.Колонки.Добавить("Категория");   
	ТаблицаТовары.Колонки.Добавить("Артикул");
	ТаблицаТовары.Колонки.Добавить("Наименование");
	ТаблицаТовары.Колонки.Добавить("Бренд");  
	ТаблицаТовары.Колонки.Добавить("Количество");
	
	ЗаполняемТаблицуТоваров(ТаблицаТовары);
	
	
	
	// Задача 1 Найти строку K-3465736 и сообщить наименование товара, 
	// если ее нет, то сообщить пользователю об этом   
	
	АртикулПоиска = ТаблицаТовары.Найти("K-3465736", "Артикул");
	Если  АртикулПоиска = Неопределено Тогда
		Сообщить("Данный артикул не найден");
	Иначе
		Сообщить("Артикул - " + АртикулПоиска.Артикул + " Наименование - "
		+ АртикулПоиска.Наименование);
	КонецЕсли;
	
	// Задание 2 Найти все холодильники бренда SAMSUNG
	
	СтруктураПоиска = Новый Структура("Бренд", "Samsung"); 
	МассивСтрок = ТаблицаТовары.НайтиСтроки(СтруктураПоиска);
	Если МассивСтрок.Количество() > 0 Тогда
		Сообщить("Найдены следующие товары:");
		Для каждого СтрокаОстаток Из МассивСтрок Цикл
			Сообщить(СтрокаОстаток.Наименование + ": (арт. [" + СтрокаОстаток.Артикул + "])");
		КонецЦикла;    
	Иначе
		Сообщить("Товар не найден");
	КонецЕсли;    
	
	// Задание 3 Найти количество товаров категории Телевизоры
	СтруктураПоиска = Новый Структура("Категория", "Телевизоры");
	МассивСтрок = ТаблицаТовары.НайтиСтроки(СтруктураПоиска);
	Если МассивСтрок.Количество() > 0 Тогда
		Сообщить("Количество товаров в категории ""Телевизоры"": " + МассивСтрок.Количество());
	Иначе
		Сообщить("Товаров этой категории нет");     
	КонецЕсли;                  
	
	// Задание 4 - Удалить из таблицы значений все строки кактегории "Электротовары"   
	СтруктураПоиска = Новый Структура("Категория", "Электротовары");
	МассивСтрокКУдалению = ТаблицаТовары.НайтиСтроки(СтруктураПоиска);
	Для каждого СтрокаКУдалению Из МассивСтрокКУдалению Цикл
		ТаблицаТовары.Удалить(СтрокаКУдалению);
	КонецЦикла;
	    
	
	// Задание 5 Получить количество товаро в каждой категории
	
	ТаблицаТовары.Свернуть("Категория", "Количество");
	Для каждого СтрокаТаблицы из ТаблицаТовары Цикл
		Сообщить(СтрокаТаблицы.Категория + " - " + СтрокаТаблицы.Количество);
	КонецЦикла;  
	        
	
	
КонецПроцедуры
